template voice(attack: float = 0.01, sustain: float = 0.7) {
    in:  voct, gate
    out: audio

    module osc : Osc
    module vca : Vca

    osc.voct <- $.voct
    $.audio  <- vca.out
    osc.sine -> vca.in
}

patch {
    module v : voice
}

template inner() {
    in:  gate
    out: audio

    module osc : Osc
    osc.sine <- $.gate
    $.audio  <- osc.sine
}

template outer() {
    in:  trig
    out: mix

    module i : inner
    i.gate <- $.trig
    $.mix  <- i.audio
}

patch {
    module top : outer
}

template bus(n: int) {
    in:  ch[n]
    out: mix

    module m : Sum(channels: <n>)
    m.in[*n] <- $.ch[*n]
    $.mix    <- m.out
}

patch {
    module b : bus(n: 4)
}
